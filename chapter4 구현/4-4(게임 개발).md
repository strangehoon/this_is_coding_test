# Python 
```python
N, M = map(int, input().split())
x, y, d = map(int, input().split())
data = []
for i in range(M):
    data.append(list(map(int, input().split())))

# 방문한 위치를 저장하기 위한 2차원 리스트
memory = [[0]*N for _ in range(M)]
memory[x][y] = 1

# 북 동 남 서
dx = [-1, 0, 1, 0]
dy = [0, 1, 0, -1]

count = 0
flag = 0
while(flag != 1):
    for i in range(4):
        nx = x+dx[(d-i-1+4)%4]
        ny = y+dy[(d-i-1+4)%4]
        # 회전한 이후 정면에 가보지 않은 육지가 존재한는 경우 이동
        if 0<=nx<=N-1 and 0<=ny<=M-1:
            if data[nx][ny]==0 and memory[nx][ny]==0:
                x = nx
                y = ny
                d = (d-i-1+4)%4
                memory[x][y] = 1
                break
        # 네 방향 모두 갈 수 없는 경우
        if i == 3:
            x += dx[(d-2+4)%4]
            y += dy[(d-2+4)%4]
            memory[x][y] = 1
            #  뒤로 갔는데 바다 일 경우 무한 루프 탈출
            if data[x][y] == 1:
                flag = 1
    if flag == 0:
        count+=1
print(count)

```



* **시간 복잡도** : O(NM)
